mysql> -- Q1: Procedure UpdateEmployeeSalary
mysql> DELIMITER //
mysql> CREATE PROCEDURE UpdateEmployeeSalary(IN emp_id INT, IN new_sal DECIMAL(10,2))
    -> BEGIN
    ->     DECLARE old_sal DECIMAL(10,2);
    ->     DECLARE emp_exists INT DEFAULT 0;
    ->
    ->     -- Check if employee exists
    ->     SELECT COUNT(*) INTO emp_exists FROM Employees WHERE employee_id = emp_id;
    ->
    ->     IF emp_exists > 0 THEN
    ->         -- Get old salary
    ->         SELECT salary INTO old_sal FROM Employees WHERE employee_id = emp_id;
    ->
    ->         -- Update salary
    ->         UPDATE Employees SET salary = new_sal WHERE employee_id = emp_id;
    ->
    ->         -- Log the change
    ->         INSERT INTO EmployeeSalaryLog (employee_id, old_salary, new_salary)
    ->         VALUES (emp_id, old_sal, new_sal);
    ->     END IF;
    -> END //

mysql> -- Q2: Function CalculateAnnualBonus
mysql> DELIMITER //
mysql> CREATE FUNCTION CalculateAnnualBonus(emp_id INT) RETURNS DECIMAL(10,2)
    -> READS SQL DATA
    -> DETERMINISTIC
    -> BEGIN
    ->     DECLARE emp_salary DECIMAL(10,2);
    ->     DECLARE emp_hire_date DATE;
    ->     DECLARE years_service INT;
    ->     DECLARE bonus DECIMAL(10,2) DEFAULT NULL;
    ->     DECLARE emp_exists INT DEFAULT 0;
    ->
    ->     -- Check if employee exists
    ->     SELECT COUNT(*) INTO emp_exists FROM Employees WHERE employee_id = emp_id;
    ->
    ->     IF emp_exists = 0 THEN
    ->         RETURN NULL;
    ->     END IF;
    ->
    ->     -- Get salary and hire_date
    ->     SELECT salary, hire_date INTO emp_salary, emp_hire_date
    ->     FROM Employees WHERE employee_id = emp_id;
    ->
    ->     -- Calculate years of service
    ->     SET years_service = TIMESTAMPDIFF(YEAR, emp_hire_date, CURDATE());
    ->
    ->     -- Calculate bonus
    ->     IF years_service < 5 THEN
    ->         SET bonus = emp_salary * 0.10;
    ->     ELSEIF years_service < 10 THEN
    ->         SET bonus = emp_salary * 0.15;
    ->     ELSE
    ->         SET bonus = emp_salary * 0.20;
    ->     END IF;
    ->
    ->     RETURN bonus;
    -> END //

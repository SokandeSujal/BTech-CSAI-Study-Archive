mysql> -- BEFORE INSERT Trigger: Validate PRN length, age >=18, fees >0
mysql> DELIMITER //
mysql> CREATE TRIGGER validate_student_insert BEFORE INSERT ON student FOR EACH ROW
    -> BEGIN
    -> DECLARE calculated_age INT;
    ->
    -> -- PRN length check
    -> IF LENGTH(NEW.prn) < 10 THEN
    -> SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'PRN must be at least 10 characters long';
    -> END IF;
    ->
    -> -- Age calculation: TIMESTAMPDIFF(YEAR, dob, CURDATE()) >=18
    -> SET calculated_age = TIMESTAMPDIFF(YEAR, NEW.dob, CURDATE());
    -> IF calculated_age < 18 THEN
    -> SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Student must be at least 18 years old';
    -> END IF;
    ->
    -> -- Fees >0
    -> IF NEW.fees <= 0 THEN
    -> SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Fees must be greater than 0';
    -> END IF;
    -> END //
